programa {
        inclua biblioteca Matematica --> mat
   funcao inicio() {
      real f, c, pol, mm

      leia(f)
      leia(pol)

      c = (5 * f - 160) / 9
      mm = pol * 25.4
     c = mat.arredondar(c, 2)
     mm = mat.arredondar(mm, 2)

      escreva("O VALOR EM CELSIUS = ", c, "\n")
      escreva("A QUANTIDADE DE CHUVA E = ", mm, "\n")
   }
}
